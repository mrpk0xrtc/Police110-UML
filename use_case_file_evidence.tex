\begin{enumerate}
    \item \textbf{جمع آوری:} در محل, یادداشت شرایط و مکان, استفاده از دستکش
    \item \textbf{بسته بندی:} مهر و موم در بسته بندی مناسب, جلوگیری از آلودگی
    \item \textbf{برچسپ گذاری:} افزودن برچسب و شناسه منحصر به فرد, اطلاعات پایه
    \item \textbf{انتقال:} وسیله نقلیه ی مناسب, حفاظت حین انتقال, تخصیص مکان امن
    \item \textbf{ثبت:} ثبت در سیستم مدارک, تایید کارشناسان
    \item \textbf{نگهداری:} رسیدگی مرتب, انبار مناسب
    \item \textbf{نابودسازی:} فقط با دسترسی مجاز, مستند شود
\end{enumerate}
